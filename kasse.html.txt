<script>
let currentItem=""
let selectedExtras = {}

// Popup öffnen
function openPopup(item){
  currentItem=item
  document.getElementById("title").innerText=item
  document.getElementById("popup").style.display="block"
  // Extras zurücksetzen
  selectedExtras = {}
  document.querySelectorAll('#popup .extra').forEach(e=>e.classList.remove('selected'))
}

// Extra auswählen
function toggleExtra(extra){
  selectedExtras[extra] = !selectedExtras[extra]
  const el = document.getElementById("extra-"+extra)
  if(selectedExtras[extra]){
    el.classList.add('selected')
  } else {
    el.classList.remove('selected')
  }
}

// Bestellung senden
function sendOrder(){
  let extras=[]
  if(selectedExtras['mayo']) extras.push("Mayo")
  if(selectedExtras['ketchup']) extras.push("Ketchup")
  if(selectedExtras['sosse']) extras.push("Pommessoße")

  let orders = JSON.parse(localStorage.getItem("orders")) || []

  // Bestellnummer berechnen
  let lastNumber = parseInt(localStorage.getItem("lastOrderNumber")) || 0
  let newNumber = lastNumber + 1
  if(newNumber > 999) newNumber = 1
  localStorage.setItem("lastOrderNumber", newNumber)
  let orderNumber = "#" + String(newNumber).padStart(3,'0')

  orders.push({
    item: currentItem,
    extras: extras,
    id: Date.now(),
    number: orderNumber
  })

  localStorage.setItem("orders", JSON.stringify(orders))

  // Popup schließen
  document.getElementById("popup").style.display="none"

  // Bestellnummer anzeigen
  document.getElementById("orderNumberText").innerText = "Bestellnummer: " + orderNumber
  document.getElementById("numberPopup").style.display="block"
}

// Abbrechen-Button Funktion
function cancelOrder(){
  // Popup schließen
  document.getElementById("popup").style.display="none"
  // Extras zurücksetzen
  selectedExtras = {}
  document.querySelectorAll('#popup .extra').forEach(e=>e.classList.remove('selected'))
}

// Bestellnummer Fenster schließen
function closeNumberPopup(){
  document.getElementById("numberPopup").style.display="none"
}
</script>
